String command;

void setup() {
  pinMode(8, OUTPUT);
  Serial.begin(9600);
  Serial.println("Enter Command: ON or OFF");
}

void loop() {
  if (Serial.available()) {
    command = Serial.readStringUntil('\n');
    command.trim();

    if (command == "ON") {
      digitalWrite(8, HIGH);
      Serial.println("Device Turned ON");
    }
    else if (command == "OFF") {
      digitalWrite(8, LOW);
      Serial.println("Device Turned OFF");
    }
    else {
      Serial.println("Invalid Command");
    }
  }
}
